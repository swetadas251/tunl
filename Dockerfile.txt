# Build stage
FROM golang:1.21-alpine AS builder

WORKDIR /app

# Copy go mod files
COPY go.mod go.sum ./
RUN go mod download

# Copy source code
COPY . .

# Build the relay binary
RUN CGO_ENABLED=0 GOOS=linux go build -o relay ./cmd/relay

# Runtime stage
FROM alpine:latest

WORKDIR /app

# Copy binary from builder
COPY --from=builder /app/relay .

# Expose port
EXPOSE 8080

# Run the relay
CMD ["./relay"]
